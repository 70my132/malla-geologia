Toca un ramo para marcarlo como cursado. Mantén presionado para ver detalles. Ramos relacionados se iluminan al pasar el mouse.

const progreso = JSON.parse(localStorage.getItem("progresoMalla") || "{}"); function guardarProgreso() { localStorage.setItem("progresoMalla", JSON.stringify(progreso)); } function toggleRamo(sIdx, rIdx) { const key = `s${sIdx}_r${rIdx}`; progreso[key] = !progreso[key]; guardarProgreso(); renderMalla(); } function mostrarDetalle(ramo) { document.getElementById("modalTitle").textContent = ramo.nombre; document.getElementById("modalContent").textContent = ramo.desc; document.getElementById("modal").classList.remove("hidden"); } function cerrarModal() { document.getElementById("modal").classList.add("hidden"); } function resaltarGrupo(grupo, entrar) { document.querySelectorAll(`[data-grupo="${grupo}"]`).forEach(el => { if (entrar) { el.classList.add("highlight"); } else { el.classList.remove("highlight"); } }); } function renderMalla() { const container = document.getElementById("malla"); container.innerHTML = ""; malla.forEach((sem, sIdx) => { const col = document.createElement("div"); col.className = "rounded-xl card p-4 shadow-lg"; const title = document.createElement("h2"); title.className = "text-lg font-semibold text-pink-700 mb-2 text-center"; title.textContent = `Semestre ${sem.semestre}`; col.appendChild(title); sem.ramos.forEach((ramo, rIdx) => { const key = `s${sIdx}_r${rIdx}`; const item = document.createElement("div"); item.className = `cursor-pointer p-1 rounded hover:bg-pink-200 transition ${progreso[key] ? 'checked' : ''}`; item.textContent = ramo.nombre; item.setAttribute("data-grupo", ramo.grupo); item.onclick = () => toggleRamo(sIdx, rIdx); item.oncontextmenu = (e) => { e.preventDefault(); mostrarDetalle(ramo); }; item.onmouseenter = () => resaltarGrupo(ramo.grupo, true); item.onmouseleave = () => resaltarGrupo(ramo.grupo, false); col.appendChild(item); }); container.appendChild(col); }); } renderMalla();
